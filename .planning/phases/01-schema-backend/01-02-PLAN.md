```python
class PersistenceBackend(ABC):
    @abstractmethod
    def insert_batch(self, rows: List[Dict[str, Any]]) -> InsertResult: ...
    @abstractmethod
    def query(self, start_ts=None, end_ts=None, event_type=None, instrument_id=None, module_id=None, limit=1000, offset=0) -> QueryResult: ...
    @abstractmethod
    def health(self) -> Dict[str, Any]: ...
    @abstractmethod
    def close(self) -> None: ...
    @abstractmethod
    def ensure_schema(self) -> bool: ...

@dataclass
class InsertResult:
    success: bool
    rows_inserted: int = 0
    error: Optional[str] = None
    def to_dict(self) -> Dict[str, Any]: ...
    @classmethod
    def from_dict(cls, d: Dict[str, Any]) -> "InsertResult": ...

@dataclass
class QueryResult:
    success: bool
    rows: List[Dict[str, Any]] = field(default_factory=list)
    error: Optional[str] = None
    def to_dict(self) -> Dict[str, Any]: ...
    @classmethod
    def from_dict(cls, d: Dict[str, Any]) -> "QueryResult": ...
```
